set %timer #scnt + 100
set %jrnl #jindex

while #true
{
  if #contname = trade_gump
  {
     click 54 44 dmc
     wait 1s
  }

  if %timer < #scnt
  {
    event macro 13 21
    set %timer #scnt + 100
  }

  finditem LKF_CKF_TXJC C_ , #backpackid
  if #findcnt > 0
  {
    wait 1s
    set %lootbag #findid
    event macro 13 56
    wait 10
    set %x 29 + #CONTPOSX
    set %y 220 + #CONTPOSY
    click %x %y dmc
    target 2s
    set #LTARGETID %lootbag
    event macro 22
    wait 1s
    if #CONTKIND = AYQ
    {
      set %x 20 + #CONTPOSX
      set %y 200 + #CONTPOSY
      click %x %y dmc
    }
    wait 1s
    set %moveItem %lootbag
    set %movestack 1
    set %findIndex 1
    repeat
    {
      finditem WN_BO %findIndex G_2
      Exevent Drag %moveItem %movestack
      wait 10
      Exevent Dropc #findid
      wait 1s
      gosub findFull
      if #RESULT
         set %findIndex %findIndex + 1
      finditem LKF_CKF_TXJC_WKR_TKR C_ , #backpackid
      set %moveItem #FINDID
      set %movestack #findstack
    }
    until #FINDCNT = 0
  }
}

sub findFull
  while #jindex >= %jrnl
   {
        set %jrnl %jrnl + 1
        scanjournal %jrnl
        if That_container_cannot in #journal
           return #true
   }
return #false
